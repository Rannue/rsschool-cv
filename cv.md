function longest(s1, s2) {
  let obj = {};
  let str = '';
  for (let i = 0; i < s1.length; i++) {
    if (!obj[s1[i]]) {
      str += s1[i];
      obj[s1[i]] = 1;
    }
  }
  for (let k = 0; k < s2.length; k++) {
    if (!obj[s2[k]]) {
      str += s2[k];
      obj[s2[k]] = 1;
    }
  }
  return str.split('').sort().join('')
}
